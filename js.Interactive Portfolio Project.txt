document.getElementById("regForm").addEventListener("submit", function(e){
    e.preventDefault();

    let name = document.getElementById("name").value;
    let email = document.getElementById("email").value;
    let pass = document.getElementById("pass").value;
    let cpass = document.getElementById("cpass").value;
    let phone = document.getElementById("phone").value;

    let valid = true;

    document.getElementById("nameErr").innerText = "";
    document.getElementById("emailErr").innerText = "";
    document.getElementById("passErr").innerText = "";
    document.getElementById("cpassErr").innerText = "";
    document.getElementById("phoneErr").innerText = "";

    if(name === ""){
        document.getElementById("nameErr").innerText = "Name required";
        valid = false;
    }

    if(email.indexOf("@") === -1 || email.indexOf(".") === -1){
        document.getElementById("emailErr").innerText = "Invalid email";
        valid = false;
    }

    if(pass.length < 6){
        document.getElementById("passErr").innerText = "Password must be 6+ characters";
        valid = false;
    }

    if(pass !== cpass){
        document.getElementById("cpassErr").innerText = "Passwords do not match";
        valid = false;
    }

    if(isNaN(phone)){
        document.getElementById("phoneErr").innerText = "Only digits allowed";
        valid = false;
    }

    if(valid){
        document.getElementById("success").innerText = "Registration Successful!";
    }
});
</script>

<script>
function analyze(){
    let text = document.getElementById("text").value.trim();

    if(text === ""){
        document.getElementById("result").innerHTML = "Please enter some text.";
        return;
    }

    let characters = text.length;
    let words = text.split(/\s+/).length;
    let reversed = text.split("").reverse().join("");

    document.getElementById("result").innerHTML = `
        <p><b>Characters:</b> ${characters}</p>
        <p><b>Words:</b> ${words}</p>
        <p><b>Reversed:</b> ${reversed}</p>
    `;
}
</script>

<script>
function updateClock(){
    let now = new Date();

    let h = now.getHours();
    let m = now.getMinutes();
    let s = now.getSeconds();

    let ampm = h >= 12 ? "PM" : "AM";
    h = h % 12 || 12;

    if(m < 10) m = "0" + m;
    if(s < 10) s = "0" + s;

    document.getElementById("clock").innerText = `${h}:${m}:${s} ${ampm}`;
}

setInterval(updateClock, 1000);
updateClock();
</script>

}

showSection("about");

    if(dark){
        body.style.background = "white";
        body.style.color = "black";
        modeBtn.innerText = "Dark Mode";

        body.style.background = "black";

// Contact Validation
function validate(){
    let name = document.getElementById("name").value;
    let email = document.getElementById("email").value;
    let msg = document.getElementById("message").value;

    nameErr.innerText = "";
    emailErr.innerText = "";
    msgErr.innerText = "";

    let ok = true;

    if(name === ""){ nameErr.innerText = "Name required"; ok = false; }
    if(email.indexOf("@") === -1 || email.indexOf(".") === -1){
        emailErr.innerText = "Invalid email"; ok = false;
    }
    if(msg.length < 10){ msgErr.innerText = "Message too short"; ok = false; }

    if(ok) alert("Form Submitted!");
